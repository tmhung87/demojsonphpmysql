28       @NonNull 
29       @Override 
30       public View getView(int position, @Nullable View convertView, @NonNull ViewGroup parent) { 
31           ViewHolder viewHolder; 
32           Tools tools = getItem(position); 
33    
34           if(convertView == null){ 
35    
36               convertView = LayoutInflater.from(context).inflate(R.layout.tools,null); 
37               viewHolder = new ViewHolder(); 
38               viewHolder.txtname = (TextView)convertView.findViewById(R.id.textViewName); 
39               viewHolder.txtprn = (TextView)convertView.findViewById(R.id.textViewPRN); 
40               viewHolder.txtsn = (TextView)convertView.findViewById(R.id.textViewSN); 
41               viewHolder.anh1 = (ImageView) convertView.findViewById(R.id.imageView); 
42               convertView.setTag(viewHolder); 
43           } else { 
44               viewHolder =(ViewHolder) convertView.getTag(); 
45           } 
46    
47           viewHolder.txtname.setText(tools.ten); 
48           viewHolder.txtprn.setText(tools.prn); 
49           viewHolder.txtsn.setText(tools.sn);

52       }
