48       private void getData(String url) { 
49           RequestQueue requestQueue = Volley.newRequestQueue(this); 
50           JsonArrayRequest jsonArrayRequest = new JsonArrayRequest(Request.Method.GET, url, null, new Response.Listener<JSONArray>() { 
51               @Override 
52               public void onResponse(JSONArray response) { 
53                   for (int i = 0; i < response.length(); i++) { 
54                       try { 
55                           JSONObject jsonObject = response.getJSONObject(i); 
56                           toolsArrayList.add(new Tools( 
57                                   jsonObject.getInt("ID"), 
58                                   jsonObject.getString("TEN"), 
59                                   jsonObject.getString("PN"), 
60                                   jsonObject.getString("SN"), 
61                                   jsonObject.getString("ANH1"), 
62                                   jsonObject.getString("ANH2"), 
63                                   jsonObject.getString("ANH3") 
64                                   )); 
65                       } catch (JSONException e) { 
66                           e.printStackTrace(); 
67                       } 
68                   } 
69                   toolsAdapter.notifyDataSetChanged(); 
70    
71               } 
72           }, new Response.ErrorListener() { 
73               @Override 
74               public void onErrorResponse(VolleyError error) { 
75                   Toast.makeText(MainActivity.this, error.toString(), Toast.LENGTH_SHORT).show(); 
76    
77               } 
78           }); 
79           requestQueue.add(jsonArrayRequest); 
80       }
